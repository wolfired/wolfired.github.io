{% highlight bash %}
function git_status() {
	cd $1
	expr $(git status --porcelain 2>/dev/null| grep "^[ M?]" | wc -l) > /dev/null
	if [[ 0 == $? ]]; then
		pwd
		git status -s
	fi
}
function for_each() {
	for dir in $1/*/; do
		git_status $dir
	done
}
